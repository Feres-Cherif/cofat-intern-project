    🧩 Component Architecture & Specifications

    💡 Angular Component Basics

    Each Angular component consists of four files:

    • .component.ts - TypeScript logic and data
    • .component.html - HTML template structure
    • .component.scss - Styling and appearance
    • .component.spec.ts - Unit tests (auto-generated)

    🧭 Navbar Component Structure

    // Generate navbar component ng generate component components/navbar // navbar.component.ts structure export class NavbarComponent { menuItems = [ { label: 'Home', route: '/' }, { label: 'About', route: '/about' }, { label: 'Services', route: '/services' }, { label: 'Contact', route: '/contact' } ]; isMobileMenuOpen = false; toggleMobileMenu() { this.isMobileMenuOpen = !this.isMobileMenuOpen; } }

    🎪 Hero Section Structure

    // Generate hero component ng generate component components/hero // hero.component.ts structure export class HeroComponent { companyName = 'COFAT'; headline = 'Leading the Future of Technology'; subtitle = 'Innovative solutions for modern business challenges'; ctaText = 'Learn More'; onCtaClick() { console.log('CTA clicked'); } }
